// src/types.ts

// Restaurant type
export interface Restaurant {
  id: string;  // Firestore document ID
  name: string;
  location: string;
}

// Menu type
export interface Menu {
  id: string; // Firestore document ID
  name: string;
  description: string;
  category: string;
}

// Menu Item type
export interface MenuItem {
  id: string; // Firestore document ID
  name: string;
  description: string;
  price: number;
  photo_url: string;
  allergies: string[];
  add_ons: string[];
}

// Branch type (represents a restaurant branch)
export interface Branch {
  id: string;  // Firestore document ID
  name: string;
  location: string;
  employees: string[]; // Array of employee IDs
}

// Employee type
export interface Employee {
  id: string; // Firestore document ID
  name: string;
  role: string;
  orders: string[]; // Array of order IDs
}

// Order type
export interface Order {
  id: string;  // Firestore document ID
  employeeId: string;
  tableId: string;
  items: MenuItem[];
}
